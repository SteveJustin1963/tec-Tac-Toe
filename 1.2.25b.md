; TEC-1 Tic-Tac-Toe for Z80 Assembly
; Uses the keypad for input and the 7-segment display for output

DISPLAY_PORT    EQU 0x01   ; Example port for display
KEYPAD_PORT     EQU 0x02   ; Example port for keypad

ORG 0x8000   ; Load address for the program
START:
    CALL INIT_BOARD  ; Initialize board state
    CALL SHOW_BOARD  ; Display initial empty board
    LD A, 1          ; Player starts first
    LD (PLAYER_TURN), A
GAME_LOOP:
    CALL SHOW_BOARD  ; Display the board
    CALL CHECK_WIN   ; Check for a winner
    JP NZ, END_GAME  ; If a winner is found, end game
    CALL CHECK_DRAW  ; Check for a draw
    JP NZ, END_GAME  ; If the board is full, end game
    LD A, (PLAYER_TURN)
    CP 1
    JP Z, PLAYER_MOVE
    JP COMPUTER_MOVE
PLAYER_MOVE:
    CALL GET_PLAYER_MOVE ; Get player input
    CALL UPDATE_BOARD    ; Update board with player's move
    LD A, 0             ; Switch turn to computer
    LD (PLAYER_TURN), A
    JP GAME_LOOP
COMPUTER_MOVE:
    CALL MAKE_COMPUTER_MOVE ; Compute AI move
    CALL UPDATE_BOARD       ; Update board with AI move
    LD A, 1                ; Switch turn to player
    LD (PLAYER_TURN), A
    JP GAME_LOOP
END_GAME:
    CALL SHOW_WINNER  ; Show game result
    CALL RESET_GAME   ; Reset for new game
    JP START

; Initialize the game board
INIT_BOARD:
    LD HL, BOARD
    LD B, 9
    LD A, '1'
INIT_LOOP:
    LD (HL), A
    INC HL
    INC A
    DJNZ INIT_LOOP
    RET

; Clear the display
CLEAR_DISPLAY:
    XOR A                    ; Clear A register
    OUT (DISPLAY_PORT), A    ; Clear display
    RET

; Show board on 7-segment display
SHOW_BOARD:
    CALL CLEAR_DISPLAY
    LD HL, BOARD
    LD B, 9
SHOW_LOOP:
    LD A, (HL)
    CALL DISPLAY_CHAR
    INC HL
    DJNZ SHOW_LOOP
    RET

; Seven segment display patterns for digits and X/O
SEGMENT_PATTERNS:
    DB 0x3F    ; 0
    DB 0x06    ; 1
    DB 0x5B    ; 2
    DB 0x4F    ; 3
    DB 0x66    ; 4
    DB 0x6D    ; 5
    DB 0x7D    ; 6
    DB 0x07    ; 7
    DB 0x7F    ; 8
    DB 0x6F    ; 9
    DB 0x77    ; A/X
    DB 0x3F    ; O

; Display character on 7-segment display
DISPLAY_CHAR:
    CP 'X'
    JR Z, DISP_X
    CP 'O'
    JR Z, DISP_O
    SUB '0'            ; Convert ASCII to number
    LD HL, SEGMENT_PATTERNS
    LD B, 0
    LD C, A
    ADD HL, BC
    LD A, (HL)         ; Get pattern
    OUT (DISPLAY_PORT), A  ; Output to display
    RET
DISP_X:
    LD A, 10           ; Offset for X pattern
    JR GET_PATTERN
DISP_O:
    LD A, 11           ; Offset for O pattern
GET_PATTERN:
    LD HL, SEGMENT_PATTERNS
    LD B, 0
    LD C, A
    ADD HL, BC
    LD A, (HL)
    OUT (DISPLAY_PORT), A
    RET

; Read keypad input
READ_KEYPAD:
    IN A, (KEYPAD_PORT)    ; Read keypad port
    AND 0x0F               ; Mask for valid keys
    ADD A, '0'             ; Convert to ASCII
    CP '0'
    JR Z, READ_KEYPAD      ; If no key, keep polling
    RET

; Get player input from keypad
GET_PLAYER_MOVE:
    CALL READ_KEYPAD
    CP '1'
    JP C, GET_PLAYER_MOVE
    CP '9'+1
    JP NC, GET_PLAYER_MOVE
    RET

; Update the board with the latest move
UPDATE_BOARD:
    SUB '1'         ; Convert ASCII number (1-9) to 0-8 offset
    LD C, A         ; Store offset in C
    LD B, 0         ; Clear B for 16-bit addition
    LD HL, BOARD    ; Load board base address
    ADD HL, BC      ; Add offset to get correct position
    
    LD A, (PLAYER_TURN)
    CP 1
    JR NZ, PLACE_O
    LD A, 'X'       ; Player's move
    JR PLACE_MARK
PLACE_O:
    LD A, 'O'       ; Computer's move
PLACE_MARK:
    LD (HL), A      ; Place the mark on the board
    RET

; AI move selection (simple AI)
MAKE_COMPUTER_MOVE:
    LD HL, BOARD
    LD B, 9
AI_LOOP:
    LD A, (HL)
    CP 'X'
    JP Z, NEXT_AI_MOVE
    CP 'O'
    JP Z, NEXT_AI_MOVE
    ; Found empty spot
    RET
NEXT_AI_MOVE:
    INC HL
    DJNZ AI_LOOP
    RET

; Check if there's a winner
CHECK_WIN:
    LD HL, WIN_CONDITIONS
    LD B, 8         ; 8 possible win conditions
CHECK_LOOP:
    PUSH HL         ; Save win conditions pointer
    LD E, (HL)      ; Load first position of trio
    INC HL
    LD D, (HL)      ; Load second position of trio
    INC HL
    LD C, (HL)      ; Load third position of trio
    
    ; Load and compare values at these positions
    LD HL, BOARD    ; Reset to board start
    ADD HL, DE      ; Get to first position
    LD A, (HL)      ; Get first value
    LD B, A         ; Store in B
    
    LD HL, BOARD
    ADD HL, DE      ; Get to second position
    LD A, (HL)
    CP B            ; Compare with first value
    JR NZ, CHECK_NEXT
    
    LD HL, BOARD
    LD E, C         ; Use third position offset
    LD D, 0
    ADD HL, DE      ; Get to third position
    LD A, (HL)
    CP B            ; Compare with first value
    JR NZ, CHECK_NEXT
    
    ; If we get here, we found a winner
    POP HL          ; Restore stack
    LD A, 1         ; Set winner found flag
    RET
    
CHECK_NEXT:
    POP HL          ; Restore win conditions pointer
    INC HL          ; Move to next trio
    INC HL
    INC HL
    DJNZ CHECK_LOOP
    XOR A           ; No winner found
    RET

; Check for a draw
CHECK_DRAW:
    LD HL, BOARD
    LD B, 9
CHECK_DRAW_LOOP:
    LD A, (HL)
    CP '1'
    JP Z, NOT_DRAW
    INC HL
    DJNZ CHECK_DRAW_LOOP
    LD A, 1
    RET
NOT_DRAW:
    XOR A
    RET

; Display game winner
SHOW_WINNER:
    CALL CLEAR_DISPLAY
    LD A, (PLAYER_TURN)
    CP 1
    JR Z, SHOW_O_WINS    ; If player turn=1, O just won
    LD A, 'X'
    CALL DISPLAY_CHAR
    RET
SHOW_O_WINS:
    LD A, 'O'
    CALL DISPLAY_CHAR
    RET

; Reset the game
RESET_GAME:
    CALL INIT_BOARD
    RET

; WIN_CONDITIONS stores winning position offsets
WIN_CONDITIONS:
    DB 0, 1, 2      ; Top row
    DB 3, 4, 5      ; Middle row
    DB 6, 7, 8      ; Bottom row
    DB 0, 3, 6      ; Left column
    DB 1, 4, 7      ; Middle column
    DB 2, 5, 8      ; Right column
    DB 0, 4, 8      ; Diagonal \
    DB 2, 4, 6      ; Diagonal /

; Board representation
BOARD:
    DS 9, ' '    ; Reserve 9 bytes for the board

; Player turn flag (1=player, 0=computer)
PLAYER_TURN:
    DB 1

END

function tictactoe()
    player_starts = true;  % Track who starts the game
    
    while true  % Main game loop for resets
        % Initialize board
        board = char('1':'9');
        game_over = false;
        
        while ~game_over
            % Display board and current starter
            display_game(board);
            if player_starts
                fprintf('Player starts this game\n\n');
            else
                fprintf('Computer starts this game\n\n');
            end
            
            % If computer starts, make its move first
            if ~player_starts
                fprintf('Computer thinking...\n');
                pause(1);
                computer_move = make_computer_move(board);
                board(computer_move) = 'O';
                display_game(board);
            end
            
            while ~game_over
                % Player move
                move = get_player_move(board);
                board(move) = 'X';
                display_game(board);
                
                % Check for player win
                if check_win(board, 'X')
                    pause(1);  % Pause to show final position
                    fprintf('\n*** YOU WIN! ***\n');
                    fprintf('\nPress any key to play again...\n');
                    pause;
                    game_over = true;
                    break;
                elseif check_draw(board)
                    pause(1);
                    fprintf('\n*** DRAW GAME! ***\n');
                    fprintf('Switching starting player for next game!\n');
                    player_starts = ~player_starts;  % Switch who starts next
                    fprintf('\nPress any key to play again...\n');
                    pause;
                    game_over = true;
                    break;
                end
                
                % Computer move
                fprintf('Computer thinking...\n');
                pause(1);
                computer_move = make_computer_move(board);
                board(computer_move) = 'O';
                display_game(board);
                
                % Check for computer win
                if check_win(board, 'O')
                    pause(1);
                    fprintf('\n*** COMPUTER WINS! ***\n');
                    fprintf('\nPress any key to play again...\n');
                    pause;
                    game_over = true;
                    break;
                elseif check_draw(board)
                    pause(1);
                    fprintf('\n*** DRAW GAME! ***\n');
                    fprintf('Switching starting player for next game!\n');
                    player_starts = ~player_starts;  % Switch who starts next
                    fprintf('\nPress any key to play again...\n');
                    pause;
                    game_over = true;
                    break;
                end
            end
        end
    end
end

function display_game(board)
    clc;
    fprintf('\n     7-Segment Tic-Tac-Toe\n\n');
    
    % For player display (first 3 seven segments)
    player_pos = board == 'X';  % Logical array of player's moves
    cpu_pos = board == 'O';     % Logical array of CPU's moves
    
    % Display top row (positions 1,2,3)
    fprintf('  ');
    for i = 1:3
        if player_pos(i)
            fprintf('-');
        else
            fprintf('x');
        end
    end
    fprintf('    ');
    % CPU's top row
    for i = 1:3
        if cpu_pos(i)
            fprintf('-');
        else
            fprintf('x');
        end
    end
    fprintf('\n');
    
    % Display middle row (positions 4,5,6)
    fprintf('  ');
    for i = 4:6
        if player_pos(i)
            fprintf('-');
        else
            fprintf('x');
        end
    end
    fprintf('    ');
    % CPU's middle row
    for i = 4:6
        if cpu_pos(i)
            fprintf('-');
        else
            fprintf('x');
        end
    end
    fprintf('\n');
    
    % Display bottom row (positions 7,8,9)
    fprintf('  ');
    for i = 7:9
        if player_pos(i)
            fprintf('-');
        else
            fprintf('x');
        end
    end
    fprintf('    ');
    % CPU's bottom row
    for i = 7:9
        if cpu_pos(i)
            fprintf('-');
        else
            fprintf('x');
        end
    end
    fprintf('\n\n');
end

function move = get_player_move(board)
    while true
        move = input('Your move (1-9): ');
        if ~isnumeric(move) || move < 1 || move > 9
            fprintf('Invalid move - please enter 1-9\n');
            continue;
        end
        if board(move) == 'X' || board(move) == 'O'
            fprintf('That position is already taken!\n');
            continue;
        end
        break;
    end
end

function move = make_computer_move(board)
    % First try to win
    move = find_winning_move(board, 'O');
    if move > 0
        return;
    end
    
    % Block player's winning move
    move = find_winning_move(board, 'X');
    if move > 0
        return;
    end
    
    % Take center if available
    if board(5) ~= 'X' && board(5) ~= 'O'
        move = 5;
        return;
    end
    
    % Take a corner with some randomness
    corners = [1 3 7 9];
    available_corners = corners(board(corners) ~= 'X' & board(corners) ~= 'O');
    if ~isempty(available_corners)
        % 80% chance to take a corner if available
        if rand < 0.8
            move = available_corners(randi(length(available_corners)));
            return;
        end
    end
    
    % Take any available spot randomly
    available = find(board ~= 'X' & board ~= 'O');
    if ~isempty(available)
        move = available(randi(length(available)));
    else
        move = -1;
    end
end

function move = find_winning_move(board, player)
    % Check all possible winning combinations
    wins = [1 2 3; 4 5 6; 7 8 9;    % Rows
            1 4 7; 2 5 8; 3 6 9;    % Columns
            1 5 9; 3 5 7];          % Diagonals
    
    move = -1;
    for i = 1:size(wins, 1)
        row = wins(i, :);
        board_row = board(row);
        % Count player's marks and empty spaces in this winning line
        player_count = sum(board_row == player);
        empty_count = sum(board_row ~= 'X' & board_row ~= 'O');
        
        % If we can win in this line
        if player_count == 2 && empty_count == 1
            % Find the empty position
            empty_pos = row(board_row ~= 'X' & board_row ~= 'O');
            move = empty_pos;
            % Add some randomness - 90% chance to take the winning move
            if rand < 0.9
                return;
            end
        end
    end
end

function won = check_win(board, player)
    wins = [1 2 3; 4 5 6; 7 8 9;    % Rows
            1 4 7; 2 5 8; 3 6 9;    % Columns
            1 5 9; 3 5 7];          % Diagonals
            
    for i = 1:size(wins, 1)
        if all(board(wins(i,:)) == player)
            won = true;
            return;
        end
    end
    won = false;
end

function draw = check_draw(board)
    draw = all(board == 'X' | board == 'O');
end
